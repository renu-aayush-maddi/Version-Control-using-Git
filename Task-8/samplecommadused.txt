#!/bin/sh

echo "Running lint check..."

# Example: check JavaScript using ESLint
npx eslint .

if [ $? -ne 0 ]; then
  echo "❌ Linting failed. Fix errors before committing."
  exit 1
fi

echo "Running tests..."

# Example: run tests
npm test

if [ $? -ne 0 ]; then
  echo "❌ Tests failed. Commit aborted."
  exit 1
fi

echo "✅ All checks passed. Proceeding with commit."
exit 0
